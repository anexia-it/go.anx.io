    go-e5e

    go-e5e is a support library to help Go developers develop Anexia e5e functions.

    Install

    With a correctly configured Go toolchain:

    1go get -u github.com/anexia-it/go-e5e

    Getting started

     1package main
 2
 3import (
 4	"runtime"
 5
 6	"github.com/anexia-it/go-e5e"
 7)
 8
 9type entrypoints struct{}
10
11func (f *entrypoints) MyEntrypoint(event e5e.Event, context e5e.Context) (*e5e.Return, error) {
12	return &e5e.Return{
13		Status: 200,
14		ResponseHeaders: map[string]string{
15			"x-custom-response-header": "This is a custom response header",
16		},
17		Data: map[string]interface{}{
18			"version": runtime.Version(),
19		},
20	}, nil
21}
22
23func main() {
24	if err := e5e.Start(&entrypoints{}); err != nil {
25		panic(err)
26	}
27}
